# 基础数据结构

## 1.1 栈

概念

- 栈是一个线性结构
- 特点是只能在某一端添加或者删除数据
- 先进后出原则

实现

```js
class Stack {
  constructor() {
    this.stack = []
  }
  push(item) {
    this.stack.push(item)
  }
  pop() {
    this.stack.pop()
  }
  peek() {
    return this.stack[this.getCount() - 1]
  }
  getCount() {
    return this.stack.length
  }
  isEmpty() {
    return this.getCount() === 0
  }
}
```

应用

> 匹配括号，可以通过栈的特性来完成

```js
// Valid Parentheses 有效的括号
// 匹配括号可是使用栈的特性来实现
/**
 * @param {string} s
 * @return {boolean}
 */

var isValid = function(s) {
  let len = s.length
  let first = s[0]
  if (len === 0) return true
  // 如果输入的括号字符串是奇数或者以右括号开头不符合
  if (len % 2 != 0 || [')', ']', '}'].indexOf(first) != -1) {
    return false
  }
  // 栈，存入第一个字符
  let stack = [first]
  const map = {
    ')': '(',
    ']': '[',
    '}': '{'
  }
  for (let i = 1; i < len; i++) {
    let stackLen = stack.length
    let top = stackLen > 0 ? stack[stackLen - 1] : null
    let now = s[i]
    if (map[now] === top) {
      stack.pop()
    } else {
      stack.push(now)
    }
  }
  return stack.length === 0
}

console.log(isValid('(){}'))
```
